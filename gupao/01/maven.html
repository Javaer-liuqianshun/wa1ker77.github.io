# maven

项目构建工具

# maven基础

优势：

convention over configuration(约定优于配置)

maven中约定的src/java项目结构、<version>4.0.0</version>等约定都在\lib\maven-model-builder-3.5.0.jar\org\apache\maven\model\pom-4.0.0.xml中(pom-4.0.0.xml俗称超级pom文件)

maven加载顺序：maven命令 -> 用户目录/.m2/setting.xml -> \conf\setting.xml

archetype：脚手架

dependencyManagement:

  1. 只能出现在父pom
  2. 版本控制
  3. 声明(子pom引用，不必指定版本号)

dependency:

  1. type：默认jar
  2. scopy：
    1. compile：编译(默认)
    2. test：测试
    3. provided：编译但不会打包(如servlet)
    4. runtime：运行时(如JDBC)
    5. system：本地一些jar

依赖传递：

子\父 compile test provided runtime
compile compile - - runtime
test test - - test
provided provided - provided provided
runtime runtime - - runtime

依赖仲裁：

  1. 最短路径原则
  2. 加载先后原则
  3. exclusions(排除包)

常用命令：

  1. package 打包
  2. install 把项目install到本地仓库
  3. deploy 把本地jar发布到远程仓库（私服）
  4. mvn clean package -U (打包时强制更新jar包)
  5. mvn dependency:tree > d.txt (查看依赖树)

常用插件

  1. findbugs/阿里巴巴约束： 静态代码检查
  2. versions：统一升级版本，使用方式：mvn versions:set -DnewVersion=1.1
  3. source：打包源代码，使用方式：mvn clean install
  4. assembly：打包zip、war、jar，使用方式：mvn clean package

作业：自定义maven插件，统计java文件数：

package com.gupao;

import org.apache.maven.plugin.AbstractMojo;
import org.apache.maven.plugin.MojoExecutionException;
import org.apache.maven.plugin.MojoFailureException;
import org.apache.maven.plugins.annotations.LifecyclePhase;
import org.apache.maven.plugins.annotations.Mojo;
import org.apache.maven.plugins.annotations.Parameter;

import java.io.File;
import java.util.ArrayList;
import java.util.List;

/**
 * @author: Order
 * @Description: 自定义maven插件
 * 实现Java类个数统计
 * @date: Created in 2018-08-02
 * @Modified:
 **/
@Mojo(name = "count-java-class", defaultPhase = LifecyclePhase.PACKAGE)
public class GupaoMojo extends AbstractMojo {

	private static List<String> fileList = new ArrayList<String>();

	/**
	 * 获取文件路径
	 */
	@Parameter
	private String basedir;


	/**
	 * 使用 mvn clean package -Dargs=.java
	 */
	@Parameter(property = "args")
	private String fileSuffix;


	public void execute() throws MojoExecutionException, MojoFailureException {
		scanFile(basedir);
		System.out.println("FilePath:" + basedir);
		System.out.println("FileSuffix:" + fileSuffix);
		System.out.println("FileTotal:" + fileList.size());
	}

	private void scanFile(String basedir) {
		File dir = new File(basedir);
		// 递归查找所有的class文件
		for (File file : dir.listFiles()) {
			if (file.isDirectory()) {
				scanFile(file.getAbsolutePath());
			} else {
				if (fileSuffix != null && !"".equals(fileSuffix)) {
					if (file.getName().endsWith(fileSuffix)) {
						fileList.add(file.getName());
					}
				}
			}
		}
	}
}


<!-- 自定义插件 - 实现Java类个数统计 -->
<plugin>
	<groupId>com.gupao</groupId>
	<artifactId>gupao-plugin</artifactId>
	<version>1.0-SNAPSHOT</version>
	<executions>
		<execution>
			<phase>package</phase>
			<goals>
				<goal>count-java-class</goal>
			</goals>
		</execution>
	</executions>
	<configuration>
		<basedir>${basedir}</basedir>
	</configuration>
</plugin>

# profile（环境切换）

<!-- 环境切换,要在build里面添加<resources></resources> 命令:mvn clean package(可换) -P test -->
<profiles>
    <profile>
        <id>dev</id>
        <properties>
            <profiles.active>dev</profiles.active>
        </properties>
        <activation>
            <activeByDefault>true</activeByDefault>
        </activation>
    </profile>
    <profile>
        <id>test</id>
        <properties>
            <profiles.active>test</profiles.active>
        </properties>
    </profile>
    <profile>
        <id>pro</id>
        <properties>
            <profiles.active>pro</profiles.active>
        </properties>
    </profile>
</profiles>

<build>
    <resources>
        <resource>
            <directory>${basedir}/src/main/resources</directory>
            <excludes>
                <exclude>conf/**</exclude>
            </excludes>
        </resource>
        <resource>
            <directory>src/main/resources/conf/${profiles.active}</directory>
        </resource>
	</resources>
</build>

# archetype（脚手架）

  • 生成一个archetype
    1. mvn archetype:create-from-project
    2. cd target/generated-sources/archetype/
    3. mvn install
  • 从archetype创建项目
    1. mvn archetype:generate -DarchetypeCatalog=local
